type Movie {

  actors: [Actor!]! @relationship(type: "ACTED_IN", direction: IN, properties: "ActedIn")

  screenTime: String

  name: String
  actedIn: [Movie!]! @relationship(type: "ACTED_IN", direction: OUT, properties: "ActedIn")
}
