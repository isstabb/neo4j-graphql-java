= Cypher Aggregations where node with Time

== Source schema

[source,graphql,schema=true]
----
type User {
  someTime: Time
  someTimeAlias: Time @alias(property: "_someTimeAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== EQUAL

.GraphQL-Query
[source,graphql]
----
{
  posts(where: {likesAggregate: {node: {someTime_EQUAL: "12:00:00"}}}) {
    content
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "12:00:00"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Post)
CALL {
	WITH this
	MATCH (this1:User)-[this0:LIKES]->(this)
	RETURN any(var2 IN collect(this1.someTime) WHERE var2 = time($param0)) AS var3
}
WITH *
WHERE var3 = true
RETURN this {
	.content
} AS this
----

'''
