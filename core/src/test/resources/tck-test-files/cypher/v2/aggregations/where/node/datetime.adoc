= Cypher Aggregations where node with DateTime

== Source schema

[source,graphql,schema=true]
----
type User {
  someDateTime: DateTime
  someDateTimeAlias: DateTime @alias(property: "_someDateTimeAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----
