= Cypher Aggregations where edge with LocalTime

== Source schema

[source,graphql,schema=true]
----
type User {
  name: String
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN, properties: "Likes")
}

interface Likes {
  someLocalTime: LocalTime
  someLocalTimeAlias: LocalTime @alias(property: "_someLocalTimeAlias")
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== EQUAL

.GraphQL-Query
[source,graphql]
----
{
  posts(where: {likesAggregate: {edge: {someLocalTime_EQUAL: "12:00:00"}}}) {
    content
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0": "12:00:00"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Post)
CALL {
	WITH this
	MATCH (this1:User)-[this0:LIKES]->(this)
	RETURN any(var2 IN collect(this0.someLocalTime) WHERE var2 = localtime($param0)) AS var3
}
WITH *
WHERE var3 = true
RETURN this {
	.content
} AS this
----

'''
