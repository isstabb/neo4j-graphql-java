= Cypher Aggregations Int

== Source schema

[source,graphql,schema=true]
----
type Movie {
  imdbRating: Int!
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== Average

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    imdbRating {
      average
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
RETURN {
	imdbRating: {
		average: avg(this.imdbRating)
	}
}
----

'''
