= Cypher Aggregations where node with String

== Source schema

[source,graphql,schema=true]
----
type User {
  name: String!
  someStringAlias: String @alias(property: "_someStringAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----
