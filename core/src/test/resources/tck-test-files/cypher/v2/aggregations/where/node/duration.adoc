== Source schema

[source,graphql,schema=true]
----
type User {
  someDuration: Duration
  someDurationAlias: Duration @alias(property: "_someDurationAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----
