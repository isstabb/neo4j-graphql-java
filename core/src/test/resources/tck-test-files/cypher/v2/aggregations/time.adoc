= Cypher Aggregations Time

== Source schema

[source,graphql,schema=true]
----
type Movie {
  createdAt: Time!
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== Max

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    createdAt {
      max
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
RETURN { createdAt: { max: max(this.createdAt) } }
----
