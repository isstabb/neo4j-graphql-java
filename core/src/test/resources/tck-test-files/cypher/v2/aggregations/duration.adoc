= Cypher Aggregations Duration

== Source schema

[source,graphql,schema=true]
----
type Movie {
  screenTime: Duration!
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== Max

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    screenTime {
      max
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
RETURN {
	screenTime: {
		max: max(this.screenTime)
	}
}
----

'''
