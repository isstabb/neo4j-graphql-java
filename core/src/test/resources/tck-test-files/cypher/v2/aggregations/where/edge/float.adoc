== Source schema

[source,graphql,schema=true]
----
type User {
  name: String
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN, properties: "Likes")
}

interface Likes {
  someFloat: Float
  someFloatAlias: Float @alias(property: "_someFloatAlias")
}
