== Source schema

[source,graphql,schema=true]
----
interface INode {
  id: ID @callback(operations: [CREATE], name: "nanoid")
}

type Product implements INode {
  id: ID
  name: String!
  genre: [Genre!]! @relationship(type: "HAS_GENRE", direction: OUT)
}

type Genre implements INode {
  id: ID
  value: String! @unique
}
----
