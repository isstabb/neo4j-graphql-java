= Cypher Union

== Source schema

[source,graphql,schema=true]
----
union Search = Movie | Genre

type Genre @auth(rules: [{operations: [READ], allow: {name: "$jwt.jwtAllowedNamesExample"}}]) {
  name: String
}

type Movie {
  title: String
  search: [Search!]! @relationship(type: "SEARCH", direction: OUT)
}
----
