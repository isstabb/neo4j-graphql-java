== Source schema

[source,graphql,schema=true]
----
type Source @auth(rules: [{operations: [DISCONNECT], allow: {id: "$jwt.sub"}}]) {
  id: ID!
  targets: [Target!]! @relationship(type: "HAS_TARGET", direction: OUT)
}

type Target {
  id: ID!
}
----
