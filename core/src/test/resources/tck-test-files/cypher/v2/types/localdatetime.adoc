= Cypher LocalDateTime

== Source schema

[source,graphql,schema=true]
----
type Movie {
  id: ID
  localDT: LocalDateTime
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== GTE Read

.GraphQL-Query
[source,graphql]
----
{
  movies(where: {localDT_GTE: "2010-08-23T13:45:33.250"}) {
    localDT
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "this_localDT_GTE": {
    "year": 2010,
    "month": 8,
    "day": 23,
    "hour": 13,
    "minute": 45,
    "second": 33,
    "nanosecond": 250000000
  }
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
WHERE this.localDT >= $this_localDT_GTE
RETURN this { .localDT } as this
----

'''

== Simple Create

.GraphQL-Query
[source,graphql]
----
mutation {
  createMovies(input: [{localDT: "1974-05-01T22:00:15.555"}]) {
    movies {
      localDT
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "this0_localDT": {
    "year": 1974,
    "month": 5,
    "day": 1,
    "hour": 22,
    "minute": 0,
    "second": 15,
    "nanosecond": 555000000
  }
}
----

.Expected Cypher output
[source,cypher]
----
CALL {
CREATE (this0:Movie)
SET this0.localDT = $this0_localDT
RETURN this0
}
RETURN 
this0 { .localDT } AS this0
----

'''
